// ═══════════════════════════════════════════════════════
// OMIOR Shipping Manager — Apps Script  (paste ALL of this)
// ═══════════════════════════════════════════════════════
function doGet(e) {
  var ss      = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data    = ss.getDataRange().getValues();
  var headers = data[0];

  // ── Case-insensitive exact column finder ──────────────
  function ci() {
    var names = Array.prototype.slice.call(arguments);
    for (var n = 0; n < names.length; n++) {
      var idx = headers.findIndex(function(h) {
        return h.toString().trim().toLowerCase() === names[n].trim().toLowerCase();
      });
      if (idx !== -1) return idx;
    }
    return -1;
  }

  // ── AWB column — tries every common variation ─────────
  function awbCol() {
    return ci('AWB Number','AWB','AWB No','AWB #','Tracking Number',
              'Tracking No','Tracking','Consignment No','Consignment',
              'Docket Number','Docket','AWB/Tracking');
  }

  // ── Courier/company column ────────────────────────────
  function courierCol() {
    return ci('Shipping Company Name','COURIER NAME','Courier Name',
              'Courier Company','Courier','Company Name','Carrier',
              'Logistics Company','Shipping Company','COURIER');
  }

  // ── Service type column ───────────────────────────────
  function svcCol() {
    return ci('Courier Service','SERVICE TYPE','Service Type','Chanel',
              'Channel','Shipping Mode','Mode','Type','Service',
              'Delivery Type','Shipment Type','SERVICE');
  }

  // ── Payment Due column ────────────────────────────────
  function payDueCol() {
    return ci('Payment Due','PAYMENT DUE FROM CUSTOMER',
              'Payment Due From Customer','PAYMENT DUE','Bill Amount',
              'Amount Due','Due Amount','Total Due','Invoice Amount');
  }

  // ── Payment Received column ───────────────────────────
  function payRcvCol() {
    return ci('Payment Received','PAYMENT RECEIVED',
              'Payment Received From Customer',
              'PAYMENT RECEIVED FROM CUSTOMER',
              'Amount Received','AMOUNT RECEIVED','Received','RECEIVED');
  }

  // ── Payment To Courier column ─────────────────────────
  function payToCourierCol() {
    return ci('Payment To Courier','PAYMENT TO COURIER',
              'PAYMENT DUE TO COURIER','Payment Due To Courier',
              'Courier Charges','COURIER CHARGES',
              'Amount To Courier','AMOUNT TO COURIER',
              'Courier Amount','Courier Fee');
  }

  // ── Paid To Courier column ────────────────────────────
  function paidToCourierCol() {
    return ci('Paid To Courier','PAID TO COURIER',
              'PAYMENT DONE TO COURIER','Payment Done To Courier',
              'COURIER PAID','Courier Paid','Courier Payment Done');
  }

  // ── Customer Name column (single full-name column) ────
  function custNameCol() {
    return ci('Customer Name','Name','Client Name','Recipient',
              'Ship To','Customer','Full Name','CUSTOMER NAME');
  }

  // ── Row-number column ─────────────────────────────────
  function rowNumCol() {
    return ci('ROW NUMBER','Row Number','Sr No','Sr. No',
              'Serial Number','Row No','Row #','S No','S.No');
  }

  // ── STATUS1 column ────────────────────────────────────
  function status1Col() {
    return ci('STATUS1','Status1','STATUS 1','Status 1',
              'WA STATUS','wa status','Whatsapp Status',
              'WHATSAPP STATUS','Trigger','TRIGGER','WA Trigger');
  }

  // ── STATUS4 column ────────────────────────────────────
  function status4Col() {
    return ci('STATUS4','Status4','STATUS 4','Status 4',
              'Payment Status','PAYMENT STATUS','Pay Status',
              'Payment Link Status','PAY LINK STATUS');
  }

  // ── findRow: priority — sheetRow (exact) > ROW# > old AWB > new AWB > customer name ──
  function findRow(upd) {
    var awbIdx  = awbCol();
    var rnIdx   = rowNumCol();
    var cnIdx   = custNameCol();
    var fnIdx   = ci('Your First Name');
    var lnIdx   = ci('Your Last Name');

    // ── LEVEL 1: sheetRow — 100% unambiguous direct address ──────────
    // sheetRow is the 1-based Google Sheet row number stamped at load time.
    // If present and valid, jump straight to that row — no searching needed.
    var sentSheetRow = parseInt(upd.sheetRow || 0);
    if (sentSheetRow >= 2 && sentSheetRow <= data.length) {
      return sentSheetRow - 1; // convert to 0-based data index
    }

    var sentRowNum  = String(upd.rowNum || '').trim();
    var sentOldAwb  = (upd.oldAwb  || '').trim();
    var sentNewAwb  = (upd.awb     || '').trim();
    var sentCust    = (upd.customerName || '').trim().toLowerCase();
    var sentFn      = (upd.firstName || '').trim().toLowerCase();
    var sentLn      = (upd.lastName  || '').trim().toLowerCase();

    for (var i = 1; i < data.length; i++) {
      var row = data[i];

      // 2. ROW NUMBER column — unique if sheet has it
      if (sentRowNum && rnIdx !== -1) {
        var rn = String(row[rnIdx] || '').trim();
        if (rn && rn === sentRowNum) return i;
      }

      // 3. Old AWB — reliable if AWB existed before edit
      if (sentOldAwb && awbIdx !== -1) {
        var ra = String(row[awbIdx] || '').trim();
        if (ra && ra === sentOldAwb) return i;
      }

      // 4. New AWB — match by the new value
      if (!sentOldAwb && sentNewAwb && awbIdx !== -1) {
        var ra2 = String(row[awbIdx] || '').trim();
        if (ra2 && ra2 === sentNewAwb) return i;
      }

      // 5. Full customer name (single-column sheet)
      if (sentCust && cnIdx !== -1) {
        var cn = String(row[cnIdx] || '').trim().toLowerCase();
        if (cn && cn === sentCust) return i;
      }

      // 6. First + Last name (separate-column sheet)
      if (sentFn && fnIdx !== -1) {
        var rf = String(row[fnIdx] || '').trim().toLowerCase();
        var rl = lnIdx !== -1 ? String(row[lnIdx] || '').trim().toLowerCase() : '';
        if (rf === sentFn && rl === sentLn) return i;
      }
    }
    return -1;
  }

  // ── Diagnostic helper ─────────────────────────────────
  function findRowDebug(upd) {
    var i = findRow(upd);
    if (i === -1) {
      var awbIdx = awbCol();
      var cnIdx  = custNameCol();
      var rnIdx  = rowNumCol();
      var sample = [];
      for (var r = 1; r < Math.min(data.length, 10); r++) {
        sample.push({
          sheetRow: r + 1,
          awb:  awbIdx !== -1 ? String(data[r][awbIdx]||'').trim() : 'n/a',
          name: cnIdx  !== -1 ? String(data[r][cnIdx] ||'').trim() : 'n/a',
          rn:   rnIdx  !== -1 ? String(data[r][rnIdx] ||'').trim() : 'n/a'
        });
      }
      return {
        found: false,
        sent: { sheetRow: upd.sheetRow||0, rowNum: upd.rowNum||'', awb: upd.awb||'', oldAwb: upd.oldAwb||'', customerName: upd.customerName||'' },
        detectedCols: { awb: headers[awbIdx]||'none', name: headers[cnIdx]||'none', rowNum: headers[rnIdx]||'none' },
        sheetSample: sample
      };
    }
    return { found: true, sheetRow: i + 1 };
  }

  // ════════════════════════════════════════
  // READ ALL ROWS (no action = initial sync)
  // ════════════════════════════════════════
  if (!e || !e.parameter || !e.parameter.action) {
    var rows = [];
    for (var i = 1; i < data.length; i++) {
      var row = {};
      headers.forEach(function(h, j) {
        row[h] = data[i][j] !== null ? String(data[i][j]) : '';
      });
      if (Object.values(row).some(function(v){ return v.trim(); })) rows.push(row);
    }
    return respond({ rows: rows });
  }

  // ════════════════════════════════════════
  // markPaid — record courier payment
  // ════════════════════════════════════════
  if (e.parameter.action === 'markPaid') {
    var updates     = JSON.parse(e.parameter.updates);
    var paidIdx     = paidToCourierCol();
    var ptcIdx      = payToCourierCol();
    if (paidIdx === -1) return respond({ ok: false, error: 'Paid-To-Courier column not found. Add a column named "Paid To Courier" to your sheet.' });
    updates.forEach(function(upd) {
      var i = findRow(upd);
      if (i !== -1) {
        ss.getRange(i + 1, paidIdx + 1).setValue(Number(upd.paidToCourier));
        if (ptcIdx !== -1 && (upd.payToCourierProvided === true || (upd.payToCourier !== undefined && upd.payToCourier !== ''))) {
          ss.getRange(i + 1, ptcIdx + 1).setValue(Number(upd.payToCourier));
        }
      }
    });
    return respond({ ok: true });
  }

  // ════════════════════════════════════════
  // markCustRcv — record customer payment received
  // ════════════════════════════════════════
  if (e.parameter.action === 'markCustRcv') {
    var updates = JSON.parse(e.parameter.updates);
    var rcvIdx  = payRcvCol();
    if (rcvIdx === -1) return respond({ ok: false, error: 'Payment Received column not found. Add a column named "Payment Received" to your sheet.' });
    updates.forEach(function(upd) {
      var i = findRow(upd);
      if (i !== -1) ss.getRange(i + 1, rcvIdx + 1).setValue(Number(upd.payRcv));
    });
    return respond({ ok: true });
  }

  // ════════════════════════════════════════
  // clearStatus1 — resend tracking trigger
  // ════════════════════════════════════════
  if (e.parameter.action === 'clearStatus1') {
    var updates = JSON.parse(e.parameter.updates);
    var s1Idx   = status1Col();
    if (s1Idx === -1) return respond({ ok: false, error: 'STATUS1 column not found. Expected column name: STATUS1, Status1, WA STATUS, or Trigger.' });
    updates.forEach(function(upd) {
      var i = findRow(upd);
      if (i !== -1) ss.getRange(i + 1, s1Idx + 1).setValue('');
    });
    return respond({ ok: true });
  }

  // ════════════════════════════════════════
  // clearStatus4 — resend payment link trigger
  // ════════════════════════════════════════
  if (e.parameter.action === 'clearStatus4') {
    var updates = JSON.parse(e.parameter.updates);
    var s4Idx   = status4Col();
    if (s4Idx === -1) return respond({ ok: false, error: 'STATUS4 column not found. Expected column name: STATUS4, Status4, Payment Status, or Pay Link Status.' });
    updates.forEach(function(upd) {
      var i = findRow(upd);
      if (i !== -1) ss.getRange(i + 1, s4Idx + 1).setValue('');
    });
    return respond({ ok: true });
  }

  // ════════════════════════════════════════
  // updateAwbCourier — save AWB + courier + service + amounts
  // ════════════════════════════════════════
  if (e.parameter.action === 'updateAwbCourier') {
    var updates  = JSON.parse(e.parameter.updates);
    var awbIdx   = awbCol();
    var coIdx    = courierCol();
    var svcIdx   = svcCol();
    var pdIdx    = payDueCol();
    var ptcIdx   = payToCourierCol();
    if (awbIdx === -1) return respond({ ok: false, error: 'AWB column not found. Expected: "AWB Number" or "AWB".' });
    var notFound = [];
    updates.forEach(function(upd) {
      var i = findRow(upd);
      if (i !== -1) {
        ss.getRange(i + 1, awbIdx + 1).setValue(upd.awb || '');
        if (coIdx  !== -1) ss.getRange(i + 1, coIdx  + 1).setValue(upd.company || '');
        if (svcIdx !== -1) ss.getRange(i + 1, svcIdx + 1).setValue(upd.svc     || '');
        if (pdIdx  !== -1 && upd.payDue > 0) ss.getRange(i + 1, pdIdx + 1).setValue(Number(upd.payDue));
        if (ptcIdx !== -1 && (upd.payToCourierProvided === true || (upd.payToCourier !== undefined && upd.payToCourier !== ''))) {
          ss.getRange(i + 1, ptcIdx + 1).setValue(Number(upd.payToCourier));
        }
      } else {
        notFound.push(findRowDebug(upd));
      }
    });
    if (notFound.length) return respond({ ok: false, error: 'Row not found. Debug: ' + JSON.stringify(notFound) });
    return respond({ ok: true });
  }

  // ════════════════════════════════════════
  // updateServiceType — change Surface ↔ Air
  // ════════════════════════════════════════
  if (e.parameter.action === 'updateServiceType') {
    var updates = JSON.parse(e.parameter.updates);
    var svcIdx  = svcCol();
    if (svcIdx === -1) return respond({ ok: false, error: 'Service Type column not found. Expected: "Courier Service", "Service Type", or "Chanel".' });
    updates.forEach(function(upd) {
      var i = findRow(upd);
      if (i !== -1) ss.getRange(i + 1, svcIdx + 1).setValue(upd.svc || '');
    });
    return respond({ ok: true });
  }

  // ════════════════════════════════════════
  // addRow — add new shipment
  // ════════════════════════════════════════
  if (e.parameter.action === 'addRow') {
    var newRow = JSON.parse(e.parameter.row);
    ss.appendRow(headers.map(function(h) { return newRow[h] !== undefined ? newRow[h] : ''; }));
    return respond({ ok: true });
  }

  return respond({ ok: false, error: 'Unknown action: ' + e.parameter.action });
}

function respond(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
